class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        vector<vector<int>> arr = matrix;
        int n = matrix.size();
        for (int i = 0; i < n; i++){
            for (int j = n - 1; j >= 0; j--){
                arr[i][j] = matrix[j][i];
            }
            reverse(arr[i].begin(), arr[i].end());
        }
        matrix = arr;
    }
};
